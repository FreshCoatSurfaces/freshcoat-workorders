function sendDailyReport() {
var date=document.getElementById(‘dr-date’).value;
var job=document.getElementById(‘dr-job’).value||’-’;
var lead=document.getElementById(‘dr-lead’).value||’-’;
var pmEmail=document.getElementById(‘dr-pmemail’).value.trim();
var start=document.getElementById(‘dr-start’).value||’-’;
var end=document.getElementById(‘dr-end’).value||’-’;
var crew=document.getElementById(‘dr-crew’).value||’-’;
var issues=document.getElementById(‘dr-issues’).value.trim()||‘None reported.’;
var notes=document.getElementById(‘dr-notes’).value.trim()||‘None.’;
var photos=document.getElementById(‘dr-photos’).value.trim()||‘None noted.’;
var tomorrow=document.getElementById(‘dr-tomorrow’).value.trim()||‘TBD.’;
if (!pmEmail) { showToast(‘Enter PM email first’); return; }
var dateStr=date?new Date(date+‘T12:00:00’).toLocaleDateString(‘en-US’,{weekday:‘short’,month:‘short’,day:‘numeric’,year:‘numeric’}):‘Today’;
var areasText=’’;
document.querySelectorAll(’#dr-area-list .dr-room-row’).forEach(function(row) {
var tog=row.querySelector(’.dr-row-toggle’);
var isOn=tog.querySelector(’.mchk’)&&tog.querySelector(’.mchk’).classList.contains(‘on’);
if (!isOn) return;
var nameEl=row.querySelector(’.dr-row-name’)||row.querySelector(‘input[type=text]’);
var name=nameEl?(nameEl.textContent||nameEl.value||‘Area’):‘Area’;
var rNotes=row.querySelector(’.dr-area-notes’)?row.querySelector(’.dr-area-notes’).value.trim():’’;
var pct=row.querySelector(’.dr-area-pct’)?row.querySelector(’.dr-area-pct’).value:’’;
areasText+=’  - ‘+name+(pct?’ [’+pct+’]’:’’)+’\n’;
if (rNotes) areasText+=’    ‘+rNotes+’\n’;
});
if (!areasText) areasText=’  No areas logged.\n’;
var paintText=’’;
document.querySelectorAll(’.paint-entry’).forEach(function(entry) {
var color=entry.querySelector(’.pe-color’)?entry.querySelector(’.pe-color’).value.trim():’-’;
var gal=entry.querySelector(’.pe-gal’)?entry.querySelector(’.pe-gal’).value.trim():’-’;
var sheen=entry.querySelector(’.pe-sheen’)?entry.querySelector(’.pe-sheen’).value.trim():’-’;
if (color||gal) paintText+=’  - ‘+color+’ | ‘+gal+’ gal | ‘+sheen+’\n’;
});
if (!paintText) paintText=’  None logged.\n’;
var subject=encodeURIComponent(’[Fresh Coat] Daily Report - ‘+job+’ - ‘+dateStr);
var body=encodeURIComponent(
‘FRESH COAT SURFACES - DAILY REPORT\n====================================\n\n’+
‘JOB: ‘+job+’\nDATE: ‘+dateStr+’\nTEAM LEAD: ‘+lead+’\nCREW ON SITE: ‘+crew+’\nHOURS: ‘+start+’ to ‘+end+’\n\n’+
‘————————————\nWORK COMPLETED TODAY\n————————————\n’+areasText+’\n’+
‘————————————\nPAINT USED\n————————————\n’+paintText+’\n’+
‘————————————\nISSUES / BLOCKERS\n————————————\n’+issues+’\n\n’+
‘————————————\nGENERAL NOTES\n————————————\n’+notes+’\n\n’+
‘————————————\nPHOTO NOTES\n————————————\n’+photos+’\n\n’+
‘————————————\nTOMORROWS PLAN\n————————————\n’+tomorrow+’\n\n’+
‘====================================\nSent via Fresh Coat Work Order System’
);
window.location.href=‘mailto:’+pmEmail+’?subject=’+subject+’&body=’+body;
showToast(‘Opening email…’);
}
